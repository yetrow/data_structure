void Split(LinkList C, LinkList &A, LinkList &B) {
LNode *p = C->next;
LNode *tailA;
LNode *tmp;
A = (LNode *)malloc(sizeof(LNode));
B = (LNode *)malloc(sizeof(LNode));
A->next = NULL;
B->next = NULL;
tailA = A; // tailA从A的头结点开始
while (p != NULL) {
// ==== 第一步：拆a节点，尾插到A ====
tmp = p;
p = p->next;
tailA->next = tmp; // 尾插
tailA = tmp;
tailA->next = NULL;
// ==== 第二步：拆b节点，头插到B ====
if (p != NULL) {
tmp = p;
p = p->next;
tmp->next = B->next; // 头插
B->next = tmp;
}
}
}
