// 向下调整函数：以 i 为根，在长度为 len 的堆中进行向下调整
void AdjustDown(int A[], int i, int len) {
    int temp = A[i]; // 暂存当前子树的根节点
    for (int k = 2*i + 1; k < len; k = 2*k + 1) { // 沿值较大的子节点向下筛选
        if (k + 1 < len && A[k] < A[k + 1]) 
            k++; // k 记录值较大的那个子节点的下标
        if (temp >= A[k]) 
            break; // 若父节点已经大于等于最大子节点，筛选结束
        else {
            A[i] = A[k]; // 将较大的子节点上提填补空位
            i = k;       // 空位下移到子节点位置，继续向下比较
        }
    }
    A[i] = temp; // 将当初暂存的根节点放入最终找到的空位
}

void HeapSort(int A[], int n) {
    // 1. 建初堆：从最后一个非叶子节点开始，逆序向上调整
    for (int i = n / 2 - 1; i >= 0; i--) 
        AdjustDown(A, i, n);
    
    // 2. 排序：根节点与堆尾交换，然后堆大小减1并重新调整根节点
    for (int i = n - 1; i > 0; i--) {
        int temp = A[0]; A[0] = A[i]; A[i] = temp; // 交换栈顶和堆尾
        AdjustDown(A, 0, i); // 将剩余的 i 个元素重新调整为大根堆
    }
}
